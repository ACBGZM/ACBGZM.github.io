class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        stack<int> s1, s2;
        while(l1!=NULL){
            s1.push(l1->val);
            l1=l1->next;
        }
        while(l2!=NULL){
            s2.push(l2->val);
            l2=l2->next;
        }
        int carry=0,sum=0;
        ListNode* head = new ListNode(0);
        while(!s1.empty() || !s2.empty() || carry!=0){
            int a = s1.empty()?0:s1.top();
            int b = s2.empty()?0:s2.top();
            sum = a + b + carry;
            ListNode* l = new ListNode(sum%10);
            carry=sum/10;

            l->next = head->next;
            head->next = l;
            
            if(!s1.empty()) s1.pop();
            if(!s2.empty()) s2.pop();        
        }
        return head->next;
    }
};

class Solution {

    int maxArea(vector<int>& height) {
        int max = 0;
        int head = 0, rear = height.size()-1;
        while(head<rear){
            int area = (rear - head) * min(height[head], height[rear]);

        }

    }

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> st;
        int maxlen = 0;
        int left = 0, i;
        for(i=0;i<s.size();i++){
            if(st.find(s[i]) == st.end()){
                st.insert(s[i]);
            }
            else{
                maxlen = max(maxlen, i - left);
                while(s[left] != s[i]){
                    st.erase(s[left]);
                    left++;
                }
                left++;
            }
        }
        maxlen = max(maxlen, i-left);
        return maxlen;
    }
};
